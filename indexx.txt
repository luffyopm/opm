/* --------------------------------------------------------------------------
   SINGLE LOGIN HANDLER (Paid + Demo forms share this)
   -------------------------------------------------------------------------- */
if ($_SERVER['REQUEST_METHOD'] === 'POST') {

    // Accept both Paid-form and Demo-form field names
    $username = $_POST['username']      ?? $_POST['demo_username'] ?? '';
    $password = $_POST['password']      ?? $_POST['demo_password'] ?? '';
    $phone    = $_POST['phone']         ?? null;                   // demo phone (optional)

    if ($username !== '' && $password !== '') {

        $sql  = "SELECT id, username, password, role_id, expiry_date, login_count, last_login
                 FROM users
                 WHERE username = ?";
        $stmt = $conn->prepare($sql);
        $stmt->bind_param("s", $username);
        $stmt->execute();
        $result = $stmt->get_result();
        $user   = $result->fetch_assoc();

        if ($user && password_verify($password, $user['password'])) {

            $today = date('Y-m-d');
            if ($user['expiry_date'] && strtotime($user['expiry_date']) < strtotime($today)) {
                $error = "Your account expired on " .
                         date('d-M-Y', strtotime($user['expiry_date'])) . ".";
            } else {
                // ── Successful login ───────────────────────────────
                session_regenerate_id(true);                // new, secure PHPSESSID

                /* ---------- UPDATE LOGIN STATISTICS ---------- */
                $updateSql = "UPDATE users SET login_count = login_count + 1, last_login = NOW() WHERE id = ?";
                $updateStmt = $conn->prepare($updateSql);
                $updateStmt->bind_param("i", $user['id']);
                $updateStmt->execute();
                $updateStmt->close();

                /* ---------- SESSION DATA ---------- */
                $_SESSION['user_id']  = $user['id'];
                $_SESSION['username'] = $user['username'];
                $_SESSION['role_id']  = $user['role_id'];
                $_SESSION['login_count'] = $user['login_count'] + 1; // Show updated count in session
                $_SESSION['last_login'] = date('Y-m-d H:i:s'); // Current timestamp
                if (!empty($phone)) {                       // demo users only
                    $_SESSION['phone'] = $phone;
                }

                header('Location: home.php');
                exit;

                /* ---------- LOGIN-LOG INSERT ---------- */
                /* ----------  SIGN-IN LOG  (non-critical)  ---------- */
                $ip   = $_SERVER['REMOTE_ADDR'] ?? null;
                $demo = $phone ?: null;

                /* try full insert first */
                $logSql = "INSERT INTO login_logs (user_id, username, phone, ip_address)
                           VALUES (?,?,?,?)";

                if ($logStmt = $conn->prepare($logSql)) {

                    $logStmt->bind_param('isss',
                        $user['id'],
                        $user['username'],
                        $demo,         // NULL for paid users
                        $ip
                    );
                    $logStmt->execute();
                    $logStmt->close();

                } else {
                    /* fallback: only user_id + username (old or trimmed table) */
                    $logSql2 = "INSERT INTO login_logs (user_id, username) VALUES (?,?)";
                    if ($logStmt2 = $conn->prepare($logSql2)) {
                        $logStmt2->bind_param('is', $user['id'], $user['username']);
                        $logStmt2->execute();
                        $logStmt2->close();
                    }
                    /* no second else → we just skip logging if even this fails */
                }
            }
        } else {
            $error = "Invalid username or password. If you don't have a Paid Account, please Click Demo Account option.";
        }
        $stmt->close();
        $conn->close();

        $error = "Please enter username and password.";
    }

<script>
  /* Phone-format helper */
  function enforcePhoneFormat(el){
    el.value = el.value.replace(/[^0-9-]/g,'');
    if(!el.value.startsWith('92-')) el.value = '92-';
    const digits = el.value.replace(/[^0-9]/g,'');
    if(digits.length > 2){
      el.value = '92-' + digits.slice(2,5) +
                 (digits.length >= 5 ? '-' : '') +
                 digits.slice(5,12);
    }
  }

  /* Video loader */
  const videoFrame = document.getElementById('youtubeVideo');
  document.querySelectorAll('.video-btn').forEach(btn=>{
    btn.addEventListener('click',()=>{
      const id = btn.dataset.video;
      videoFrame.src = `https://www.youtube.com/embed/${id}?rel=0&autoplay=1`;
    });
  });
  document.getElementById('videoModal')
          .addEventListener('hidden.bs.modal',()=> videoFrame.src = '');
</script>
