<script>
function updateWishlist(itemId, el) {
    $.ajax({
        url: "app/wishlist_action.php",
        type: "POST",
        dataType: "json",
        data: { item_id: itemId },
        success: function (response) {

            if (response.status === "added") {
                $(el).find("i").addClass("text-danger");
                updateWishlistCount(response.count);
            }

            else if (response.status === "removed") {
                $(el).find("i").removeClass("text-danger");
                updateWishlistCount(response.count);
            }

            else if (response.status === "login") {
                alert(response.message);

                // Redirect to login page
                window.location.href = "app/sign-in.php";
            }

            else {
                alert(response.message);
            }
        },
        error: function () {
            alert("Something went wrong. Please try again.");
        }
    });
}

function updateWishlistCount(count) {
    $("#wishlist-count").text(count);
    $("#wishlist-count-footer").text(count);
}

// Load wishlist count on page load
$(document).ready(function () {
    $.getJSON("app/wishlist_count.php", function (data) {
        updateWishlistCount(data.count);
    });
});
</script>

                    $('#qty2-' + prefix + variantId).text(1);
                    updateCartCount();
                });
            }

            function updateCart2(variantId, action, section = '') {
                const prefix = section ? section + '-' : '';

                $.post('app/cart.php', { action: action, variant_id: variantId }, function () {
                    let qtySpan = $('#qty2-' + prefix + variantId);
                    let qty = parseInt(qtySpan.text());

                    if (action === 'increase') {
                        qty++;
                    } else if (action === 'decrease') {
                        qty--;
                    }

                    if (qty <= 0) {
                        $('#qty-box2-' + prefix + variantId).addClass('d-none');
                        $('#cart-box2-' + prefix + variantId).removeClass('d-none');
                        qty = 0;
                    }

                    qtySpan.text(qty);
                    updateCartCount();
                });
            }

            function updateCartCount() {
                $.post('app/cart.php', { action: 'count' }, function (count) {
                    $('#cart-count').text(count);
                    $('#cart-count1').text(count);
                });
            }

            // Initialize cart count on load
            $(document).ready(function () {
                updateCartCount();
            });
        </script>

<script>
function loadProductModal(itemId) {
    $('#product-view .modal-content').html('<div class="text-center p-4">Loading...</div>');
    $.ajax({
        url: 'app/get_product_modal.php',
        data: { id: itemId },
        method: 'GET',
        success: function (data) {

            $('#product-view .modal-content').html(data);

            // ⚠️ IMPORTANT: Wait for DOM update
            setTimeout(function () {

                // Destroy if already initialized
                if ($('.preview-slider').hasClass('slick-initialized')) {
                    $('.preview-slider').slick('unslick');
                }
                if ($('.thumb-slider').hasClass('slick-initialized')) {
                    $('.thumb-slider').slick('unslick');
                }

                // Initialize MAIN slider
                $('.preview-slider').slick({
                    slidesToShow: 1,
                    slidesToScroll: 1,
                    arrows: true,
                    fade: false,
                    asNavFor: '.thumb-slider'
                });

                // Initialize THUMB slider
                $('.thumb-slider').slick({
                    slidesToShow: 5,
                    slidesToScroll: 1,
                    asNavFor: '.preview-slider',
                    dots: false,
                    focusOnSelect: true,
                    centerMode: false
                });

            }, 200); // small delay for safety
        },
        error: function () {
            $('#product-view .modal-content').html(
                '<div class="text-danger p-4 text-center">Failed to load product</div>'
            );
        }
    });
}
